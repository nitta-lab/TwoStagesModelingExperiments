「仲介者オブジェクトへの依存」実行後 

    public class Main {
        public static void main(String[] args) {
     		this.a = new A();   // value = 0;
            this.d = new D(a);
            this.c = new C(d);
            this.b = new B(a);

            a.setA(10);
            System.out.println(b.getValue());  // out: 20
            System.out.println(c.getValue());  // out: 11
        }
    }

    public class A {
        private int value = 0;

        public A() {
        }

        public int getValue() {
            return this.value;
        }
        public void setA(int value) {
            this.value = value;
        }
    }

    public class D {
        private A a;

        public D(A a) {
            this.a = a;
        }

        public int getAValue() {
            return this.a.getValue();
        }
    }

    public class C {
        private D d;

        public C(D d) {
            this.d = d;
        }

        public int getValue() {
            return (this.d.getAValue()+1);
        }
    }

    public class B {
        private A a;

        public B(A a) {
            this.a = a;
        }

        public int getValue() {
            return (this.a.getValue()*2);
        }
    }
